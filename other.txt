## The #1 likely cause (common with `ImageFolder`): **different `class_to_idx` for train vs val**

If your `get_vggface_dataloaders()` creates **separate datasets** via `torchvision.datasets.ImageFolder(train_dir)` and `ImageFolder(val_dir)`, each one builds its own `class_to_idx` mapping from folder names. If the folder sets/order differ, label “123” in train is not the same person as label “123” in val.

✅ Fix: build a **single shared class\_to\_idx** (from train), then apply it to val/test (or remap val labels into train’s index space).

VGGFace2 is often used with:

-   **Train identities** (thousands) for training classification/embeddings
    
-   **Test identities** (a different set) for verification/identification benchmarks

class CheXpertMultiLabelDataset(Dataset):  
    """  
    Multi-label CheXpert dataset (5 labels).  
    Returns:  
      image: Tensor \[3, 224, 224\]  
      target: FloatTensor \[5\] with 0/1 values  
    """  
  
    def \_\_init\_\_(self, csv\_file, root\_dir, transform\=None, drop\_uncertain\=True):  
        self.root\_dir \= root\_dir  
        self.transform \= transform  
  
        df \= pd.read\_csv(csv\_file)  
  
        \# Standard 5 labels (same as you used)  
        self.pathologies \= \['Atelectasis', 'Cardiomegaly', 'Consolidation', 'Edema', 'Pleural Effusion'\]  
  
        \# Handle uncertainty and NaNs  
        df \= df.fillna(0.0)  
  
        \# CheXpert uses -1 for uncertain.  
        \# For baseline we map uncertain -> 0 (or you could treat -1 as 1 depending on protocol).  
        if drop\_uncertain:  
            df\[self.pathologies\] \= df\[self.pathologies\].replace(\-1.0, 0.0)  
        else:  
            \# If you want to keep -1, you'd need a mask and custom loss.  
            df\[self.pathologies\] \= df\[self.pathologies\].replace(\-1.0, 0.0)  
  
        \# Build paths (CSV often contains 'CheXpert-v1.0-small/...')  
        self.image\_paths \= df\['Path'\].apply(lambda x: x.replace('CheXpert-v1.0-small/', '')).tolist()  
  
        \# Targets as 0/1 float matrix  
        targets \= df\[self.pathologies\].values.astype('float32')  
        targets \= (targets \> 0.5).astype('float32')  \# enforce 0/1  
        self.targets \= torch.from\_numpy(targets)  
  
    def \_\_len\_\_(self):  
        return len(self.image\_paths)  
  
    def \_\_getitem\_\_(self, idx):  
        img\_path \= os.path.join(self.root\_dir, self.image\_paths\[idx\])  
        image \= Image.open(img\_path).convert('RGB')  
  
        if self.transform:  
            image \= self.transform(image)  
  
        target \= self.targets\[idx\]  \# FloatTensor \[5\]  
        return image, target

@torch.no\_grad()  
def multilabel\_metrics\_from\_logits(logits, targets, threshold\=0.5):  
    """  
    logits:  \[B, C\]  
    targets: \[B, C\] float {0,1}  
    Returns:  
      label\_acc: average per-label accuracy over all labels and samples  
      example\_acc: exact match accuracy per sample (very strict)  
      macro\_f1: average F1 across classes  
    """  
    probs \= torch.sigmoid(logits)  
    preds \= (probs \>= threshold).float()  
  
    \# label accuracy (micro over labels)  
    label\_acc \= (preds \== targets).float().mean().item()  
  
    \# exact match accuracy (all labels correct)  
    example\_acc \= (preds.eq(targets).all(dim\=1)).float().mean().item()  
  
    \# macro F1  
    eps \= 1e-8  
    f1s \= \[\]  
    for c in range(targets.size(1)):  
        p \= preds\[:, c\]  
        t \= targets\[:, c\]  
        tp \= (p \* t).sum()  
        fp \= (p \* (1 \- t)).sum()  
        fn \= ((1 \- p) \* t).sum()  
        precision \= tp / (tp + fp + eps)  
        recall \= tp / (tp + fn + eps)  
        f1 \= (2 \* precision \* recall) / (precision + recall + eps)  
        f1s.append(f1.item())  
    macro\_f1 \= sum(f1s) / len(f1s)  
  
    return label\_acc, example\_acc, macro\_f1

def train\_epoch\_singlelabel(model, loader, criterion, optimizer, device, epoch):  
    model.train()  
    running\_loss \= 0.0  
    correct \= 0  
    total \= 0  
  
    pbar \= tqdm(loader, desc\=f"Epoch {epoch} \[Train\]")  
    for inputs, targets in pbar:  
        inputs, targets \= inputs.to(device), targets.to(device)  
  
        optimizer.zero\_grad()  
        outputs \= model(inputs)  
        loss \= criterion(outputs, targets)  
        loss.backward()  
        optimizer.step()  
  
        running\_loss += loss.item() \* inputs.size(0)  
        \_, predicted \= outputs.max(1)  
        total += targets.size(0)  
        correct += predicted.eq(targets).sum().item()  
  
        pbar.set\_postfix({'loss': loss.item(), 'acc': 100.0 \* correct / total})  
  
    epoch\_loss \= running\_loss / total  
    epoch\_acc \= 100.0 \* correct / total  
    return epoch\_loss, epoch\_acc

@torch.no\_grad()  
def multilabel\_metrics\_from\_logits(logits, targets, threshold\=0.5):  
    probs \= torch.sigmoid(logits)  
    preds \= (probs \>= threshold).float()  
  
    label\_acc \= (preds \== targets).float().mean().item()  
    example\_acc \= (preds.eq(targets).all(dim\=1)).float().mean().item()  
  
    eps \= 1e-8  
    f1s \= \[\]  
    for c in range(targets.size(1)):  
        p \= preds\[:, c\]  
        t \= targets\[:, c\]  
        tp \= (p \* t).sum()  
        fp \= (p \* (1 \- t)).sum()  
        fn \= ((1 \- p) \* t).sum()  
        precision \= tp / (tp + fp + eps)  
        recall \= tp / (tp + fn + eps)  
        f1 \= (2 \* precision \* recall) / (precision + recall + eps)  
        f1s.append(f1.item())  
    macro\_f1 \= sum(f1s) / len(f1s)  
    return label\_acc, example\_acc, macro\_f1  
  
  
def train\_epoch\_multilabel(model, loader, criterion, optimizer, device, epoch):  
    model.train()  
    running\_loss \= 0.0  
    total\_samples \= 0  
  
    running\_label\_acc \= 0.0  
    running\_example\_acc \= 0.0  
    running\_macro\_f1 \= 0.0  
    metric\_batches \= 0  
  
    pbar \= tqdm(loader, desc\=f"Epoch {epoch} \[Train\]")  
    for inputs, targets in pbar:  
        inputs \= inputs.to(device)  
        targets \= targets.to(device)  
  
        optimizer.zero\_grad()  
        logits \= model(inputs)  
        loss \= criterion(logits, targets)  
        loss.backward()  
        optimizer.step()  
  
        bs \= inputs.size(0)  
        running\_loss += loss.item() \* bs  
        total\_samples += bs  
  
        label\_acc, example\_acc, macro\_f1 \= multilabel\_metrics\_from\_logits(logits.detach(), targets.detach())  
        running\_label\_acc += label\_acc  
        running\_example\_acc += example\_acc  
        running\_macro\_f1 += macro\_f1  
        metric\_batches += 1  
  
        pbar.set\_postfix({  
            'loss': loss.item(),  
            'label\_acc': f"{100.0 \* (running\_label\_acc / metric\_batches):.1f}%",  
            'macro\_f1': f"{(running\_macro\_f1 / metric\_batches):.3f}"  
        })  
  
    epoch\_loss \= running\_loss / total\_samples  
    epoch\_label\_acc \= 100.0 \* (running\_label\_acc / metric\_batches)  
    epoch\_example\_acc \= 100.0 \* (running\_example\_acc / metric\_batches)  
    epoch\_macro\_f1 \= (running\_macro\_f1 / metric\_batches)  
    return epoch\_loss, epoch\_label\_acc, epoch\_example\_acc, epoch\_macro\_f1

@torch.no\_grad()  
def multilabel\_metrics\_from\_logits(logits, targets, threshold\=0.5):  
    """  
    logits:  \[B, C\]  
    targets: \[B, C\] float {0,1}  
    Returns:  
      label\_acc: average per-label accuracy over all labels and samples (inflated w/ imbalance)  
      example\_acc: exact match accuracy per sample (very strict)  
      macro\_f1: average F1 across classes  
    """  
    probs \= torch.sigmoid(logits)  
    preds \= (probs \>= threshold).float()  
  
    label\_acc \= (preds \== targets).float().mean().item()  
    example\_acc \= (preds.eq(targets).all(dim\=1)).float().mean().item()  
  
    eps \= 1e-8  
    f1s \= \[\]  
    for c in range(targets.size(1)):  
        p \= preds\[:, c\]  
        t \= targets\[:, c\]  
        tp \= (p \* t).sum()  
        fp \= (p \* (1 \- t)).sum()  
        fn \= ((1 \- p) \* t).sum()  
  
        precision \= tp / (tp + fp + eps)  
        recall \= tp / (tp + fn + eps)  
        f1 \= (2 \* precision \* recall) / (precision + recall + eps)  
        f1s.append(f1.item())  
  
    macro\_f1 \= sum(f1s) / len(f1s)  
    return label\_acc, example\_acc, macro\_f1

def train\_epoch\_singlelabel(model, loader, criterion, optimizer, device, epoch):  
    model.train()  
    running\_loss \= 0.0  
    correct \= 0  
    total \= 0  
  
    pbar \= tqdm(loader, desc\=f"Epoch {epoch} \[Train\]")  
    for inputs, targets in pbar:  
        inputs, targets \= inputs.to(device), targets.to(device)  
  
        optimizer.zero\_grad()  
        outputs \= model(inputs)  
        loss \= criterion(outputs, targets)  
        loss.backward()  
        optimizer.step()  
  
        running\_loss += loss.item() \* inputs.size(0)  
        \_, predicted \= outputs.max(1)  
        total += targets.size(0)  
        correct += predicted.eq(targets).sum().item()  
  
        pbar.set\_postfix({'loss': loss.item(), 'acc': 100.0 \* correct / total})  
  
    epoch\_loss \= running\_loss / total  
    epoch\_acc \= 100.0 \* correct / total  
    return epoch\_loss, epoch\_acc

-   Train MacroF1 creeps **0.326 → ~0.379** and then stalls.
    
-   Val MacroF1 swings: **0.322 ↔ 0.406**.
    

This is normal because:

1.  **MacroF1 is threshold-sensitive** (0.5 cutoff can randomly hurt/help rare classes)
    
2.  Some classes are _extremely_ rare (your `pos_weight` shows Consolidation is ~14× rarer than negatives), so small changes in predicted positives swing F1 a lot.
    
3.  With CosineAnnealing, LR changes can cause “good/bad” epochs.
    

### The “LabelAcc spikes to 74% while F1 drops” is expected

Example:

-   Epoch 9 val: **LabelAcc 74.17%** but **MacroF1 0.322**  
    That usually means the model is predicting _fewer positives_ (so it gets lots of negatives right → higher label acc) but misses positives → F1 drops.
    

That’s why saving by **MacroF1** is the right call.

### Drop-in patch: threshold sweep on validation

@torch.no\_grad()  
def multilabel\_macro\_f1\_at\_threshold(logits, targets, threshold: float):  
    probs \= torch.sigmoid(logits)  
    preds \= (probs \>= threshold).float()  
  
    eps \= 1e-8  
    f1s \= \[\]  
    for c in range(targets.size(1)):  
        p \= preds\[:, c\]  
        t \= targets\[:, c\]  
        tp \= (p \* t).sum()  
        fp \= (p \* (1 \- t)).sum()  
        fn \= ((1 \- p) \* t).sum()  
        precision \= tp / (tp + fp + eps)  
        recall \= tp / (tp + fn + eps)  
        f1 \= (2 \* precision \* recall) / (precision + recall + eps)  
        f1s.append(f1.item())  
    return sum(f1s) / len(f1s)

## The fix (use this loader for VGGFace2 classification)

If you want **classification accuracy**, you must validate on the **same identity set**. The simplest fix is: **split the train folder into train/val** (single shared mapping).

Replace your `get_vggface_dataloaders()` with this version:

Run

def get\_vggface\_dataloaders(data\_dir\='./data/VGGFace2', batch\_size\=64, val\_split\=0.1, num\_workers\=4):  
    mean \= (0.485, 0.456, 0.406)  
    std  \= (0.229, 0.224, 0.225)  
  
    train\_transform \= transforms.Compose(\[  
        transforms.Resize(256),  
        transforms.RandomResizedCrop(224),  
        transforms.RandomHorizontalFlip(),  
        transforms.ToTensor(),  
        transforms.Normalize(mean, std)  
    \])  
  
    val\_transform \= transforms.Compose(\[  
        transforms.Resize(256),  
        transforms.CenterCrop(224),  
        transforms.ToTensor(),  
        transforms.Normalize(mean, std)  
    \])  
  
    train\_dir \= os.path.join(data\_dir, 'train')  
    if not os.path.exists(train\_dir):  
        raise FileNotFoundError(f"Missing {train\_dir}")  
  
    \# ONE ImageFolder => ONE shared class\_to\_idx  
    full\_train \= datasets.ImageFolder(root\=train\_dir, transform\=train\_transform)  
  
    val\_size \= int(len(full\_train) \* val\_split)  
    train\_size \= len(full\_train) \- val\_size  
  
    g \= torch.Generator().manual\_seed(42)  
    train\_subset, val\_subset \= random\_split(full\_train, \[train\_size, val\_size\], generator\=g)  
  
    \# give val deterministic transform but same mapping  
    full\_train\_valview \= datasets.ImageFolder(root\=train\_dir, transform\=val\_transform)  
    val\_subset.dataset \= full\_train\_valview  
  
    train\_loader \= DataLoader(train\_subset, batch\_size\=batch\_size, shuffle\=True, num\_workers\=num\_workers, pin\_memory\=True)  
    val\_loader   \= DataLoader(val\_subset, batch\_size\=batch\_size, shuffle\=False, num\_workers\=num\_workers, pin\_memory\=True)  
  
    test\_loader \= val\_loader  \# baseline  
    return train\_loader, val\_loader, test\_loader, full\_train
